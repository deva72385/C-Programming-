#include <stdio.h>

int main() {
	int a,b,c;
	scanf("%d",&a);
	scanf("%d",&b);
	scanf("%d",&c);
	
     if (a>=b&&a>=c){
         printf("Largest:%d\n",a);
     }
     else if (b>=c&&b>=a){
         printf("Largest:%d\n",b);
     }
     else
     printf("Largest:%d\n",c);
     return 0;
}
